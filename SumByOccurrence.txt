package Leetcode;
import java.util.*;
//SumByOccurrence
public class Z15 {
    public static void main(String[] args) {
        String input = "1223334444";
        Map<Character, Integer> countMap = new HashMap<>();
        Map<Character, Integer> sumMap = new HashMap<>();

        // Count occurrences and sum digits
        for (char c : input.toCharArray()) {
            countMap.put(c, countMap.getOrDefault(c, 0) + 1);
            sumMap.put(c, sumMap.getOrDefault(c, 0) + (c - '0'));
        }

        // Group sums by occurrence count
        Map<Integer, Integer> resultMap = new TreeMap<>();
        for (char c : countMap.keySet()) {
            int freq = countMap.get(c);
            int sum = sumMap.get(c);
            resultMap.put(freq, resultMap.getOrDefault(freq, 0) + sum);
        }

        // Format output
        StringBuilder output = new StringBuilder();
        for (Map.Entry<Integer, Integer> entry : resultMap.entrySet()) {
            output.append(entry.getKey()).append("-").append(entry.getValue()).append(" ");
        }

        System.out.println("Output: " + output.toString().trim());
    }
}
